set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED TRUE)

include(cmake/docker.cmake)

add_subdirectory(ext)

add_subdirectory(test)

add_parser_docker_targets()
